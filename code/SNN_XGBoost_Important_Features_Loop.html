def rmse(predictions, targets):
    differences = predictions - targets                       #the DIFFERENCEs.
    differences_squared = differences ** 2                    #the SQUAREs of ^
    mean_of_differences_squared = differences_squared.mean()  #the MEAN of ^
    rmse_val = np.sqrt(mean_of_differences_squared)           #ROOT of ^
    return rmse_val
def evaluate(model, test_features, test_labels):
    predictions = model.predict(test_features)
    errors = abs(predictions - test_labels)
    mape = 100 * np.mean(errors / test_labels)
    accuracy = 100 - mape
    print('Average Error: {:0.4f} degrees.'.format(np.mean(errors)))
    print('Accuracy = {:0.2f}%.'.format(accuracy))
    return accuracy

def mean_absolute_percentage_error(y_true, y_pred): 
    y_true, y_pred = np.array(y_true), np.array(y_pred)
    return np.mean(np.abs((y_true - y_pred) / y_true)) * 100

def update_dict(x_dict, num_inputs, learning_rate, num_HL_nrns, HL1_ac_fn, HL2_ac_fn):
    x_dict['num_inputs'] = num_inputs
    x_dict['learning_rate'] = learning_rate
    x_dict['num_HL_nrns'] = num_HL_nrns
    x_dict['HL1_ac_fn'] = HL1_ac_fn
    x_dict['HL2_ac_fn'] = HL2_ac_fn
    x_dict['n_components'] = 13

for num_HL_nrns in range(1,HL_1_nodes_max+1):
    if elapsed_time is not None:
        elapsed_time = elapsed_time*(HL_1_nodes_max-num_HL_nrns)
        print(f"\tTime Remaining: {int((elapsed_time//60)//60)}:{int((elapsed_time//60)%60)}:{int(elapsed_time%60)}", end="")
        print(f'\tETC: {time.strftime("%I:%Mp", time.localtime(time.time()+elapsed_time))}')
    start_time = time.process_time()

    for HL1_ac_fn in range(len(activation_list)):
        for HL2_ac_fn in range(len(activation_list)):
            for loss_fn in range(len(loss_list)):
                model = Sequential([
                    Dense(num_HL_nrns, activation = activation_list[HL1_ac_fn], input_shape=(X_train.shape[1],)),
                    Dense(1, activation = activation_list[HL2_ac_fn])
                ])
                model.compile(loss = loss_list[loss_fn], optimizer='adam', metrics = metrics)
                history = model.fit(X_train, Y_train, batch_size=128, verbose=False, epochs=100, validation_data=(X_test, Y_test))
                if history.history.get('root_mean_squared_error')[-1] <= rmse_dict['root_mean_squared_error']:
                    rmse_dict['root_mean_squared_error'] = history.history.get('root_mean_squared_error')[-1]
                    update_dict(rmse_dict, num_inputs, learning_rate, num_HL_nrns, activation_list[HL1_ac_fn], activation_list[HL2_ac_fn])
                if history.history.get('mean_absolute_percentage_error')[-1] <= mape_dict['mean_absolute_percentage_error']:
                    mape_dict['mean_absolute_percentage_error'] = history.history.get('mean_absolute_percentage_error')[-1]
                    update_dict(mape_dict, num_inputs, learning_rate, num_HL_nrns, activation_list[HL1_ac_fn], activation_list[HL2_ac_fn])
                if history.history.get('mean_absolute_error')[-1] <= mae_dict['mean_absolute_error']:
                    mae_dict['mean_absolute_error'] = history.history.get('mean_absolute_error')[-1]
                    update_dict(mae_dict, num_inputs, learning_rate, num_HL_nrns, activation_list[HL1_ac_fn], activation_list[HL2_ac_fn])
    elapsed_time = time.process_time() - start_time
